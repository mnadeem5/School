3)	public void decreaseRed()
  	{
    		Pixel[] pixelArray = this.getPixels();
    		Pixel pixelObj = null;
    		double value;
    		for(int index=0; index < pixelArray.length; index++)
    		{
      			pixelObj = pixelArray[index];
      			value = pixelObj.getRed()*0.8;
      			pixelObj.setRed((int)value);
    		}
  	}

	pictureObj.explore();

	pictureObj.explore();
